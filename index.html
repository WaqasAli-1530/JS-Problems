Task:

Write a JavaScript program to calculate the days left before Christmas.

Solution: 

            function days_left()
                {
                    const d_now = new Date();
                    const d_chris = new Date(2023,11,25);
                    let mili_seconds = d_chris - d_now;
                    let seconds = mili_seconds / 1000;
                    let minutes = seconds / 60;
                    let hour = minutes / 60;
                    let day = hour / 24;
                    return "Days left from Christmis are: " + Math.floor(day) 
                }
                
                console.log(days_left())

Task:

Write a JavaScript program to create another string by adding "Py" in front of a given string. If the given string begins with "Py" return the original string.

Solution: 

            function py_str(str)
                {
                    let flag = str.startsWith("Py");
                    if (flag)
                        return str;
                    else
                        return "Py" + str;
                }
                console.log(py_str("Pymysql"))
                console.log(py_str("Hello"))

Task:

Write a JavaScript program to change the capitalization of all letters in a given string.

Solution: 

            function change_capt(str)
                {
                    let new_str = "";
                    for(let x of str)
                    {
                        if(x.charCodeAt() >= 65 && x.charCodeAt() <= 90)
                            new_str += x.toLowerCase();
                        else if(x.charCodeAt() >= 97 && x.charCodeAt() <= 122)
                            new_str += x.toUpperCase();
                        else
                            new_str += x;
                    }
                    return new_str;
                }
                console.log(change_capt("I Love Pakistan "))

Task:

Write a JavaScript program to compute the sum of all the digits that occur in a given string.

Solution: 

            function sum_string_digits(str)
                {
                    let sum = 0;
                    for(let x of str)
                    {
                        if(x.charCodeAt() >= 48 && x.charCodeAt() <= 57)
                            sum += parseInt(x);
                    }
                    return sum;
                }
                console.log(sum_string_digits("123Hammad456"))

Task:

Write a JavaScript program to break an URL address and put its parts into an array.

Solution: 

            function break_url(url)
                {
                    const url_array = [];
                    const tok1 = url.split(":");
                    url_array.push(tok1[0]);
                    const tok2 = tok1[1].split("/");
                    for(let x of tok2)
                    {
                        if(x != "")
                        {
                            if(url_array[2] != undefined)
                                url_array[2] += "/" + x;
                            else
                                url_array.push(x);
                        }
                           
                    }
                    return url_array
                }
                console.log(break_url("https://facebook.com/Home/"))
                console.log(break_url("https://facebook.com/Home/reels"))

Task:

Write a JavaScript program to sort the strings of a given array of strings in order of increasing length.

Solution: 

            function sort(array)
                {
                    return array.sort(function(a,b){return a.length - b.length});
                }
